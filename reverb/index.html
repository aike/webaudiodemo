var convolver = ctx.createConvolver();
var xhr = new XMLHttpRequest();
xhr.open("GET", "ir.wav", true);
xhr.responseType = "arraybuffer";
xhr.onload = function() {
    ctx.decodeAudioData(xhr.response,function(buf){
        convolver.buffer = buf;
    }, function(){});
};
xhr.send();
